2. Control Flow

2.1 Conditional Statements

# if-elif-else
x = 15
if x > 20:
    print("big")
elif x > 10:
    print("medium")    # Output: medium
else:
    print("small")

2.2 Loops

# for loop with range
for i in range(5):       # 0, 1, 2, 3, 4
    print(i, end=" ")
# Output: 0 1 2 3 4

# range(start, stop, step)
for i in range(2, 10, 3):  # 2, 5, 8
    print(i, end=" ")
# Output: 2 5 8

# while loop with break/continue
i = 0
while i < 10:
    i += 1
    if i == 3:
        continue    # skip 3
    if i == 7:
        break       # stop at 7
    print(i, end=" ")
# Output: 1 2 4 5 6

# for-else (else runs if no break)
for i in range(5):
    if i == 10:
        break
else:
    print("completed")  # Output: completed

3.1 List Comprehensions

# Basic
squares = [x**2 for x in range(6)]
print(squares)  # [0, 1, 4, 9, 16, 25]

# With condition
evens = [x for x in range(10) if x % 2 == 0]
print(evens)  # [0, 2, 4, 6, 8]

# Nested
matrix = [[i*3+j for j in range(3)] for i in range(3)]
print(matrix)  # [[0,1,2], [3,4,5], [6,7,8]]

# Flatten
flat = [x for row in matrix for x in row]
print(flat)  # [0, 1, 2, 3, 4, 5, 6, 7, 8]

6. Functions

# Default arguments
def greet(name, greeting="Hello"):
    return f"{greeting}, {name}!"

print(greet("Alice"))           # Hello, Alice!
print(greet("Bob", "Hi"))       # Hi, Bob!

# *args and **kwargs
def add(*args):
    return sum(args)

print(add(1, 2, 3))    # 6

def info(**kwargs):
    for k, v in kwargs.items():
        print(f"{k}: {v}")

info(name="Alice", age=25)
# name: Alice
# age: 25

# Lambda functions
square = lambda x: x**2
print(square(5))    # 25

# Map, Filter, Reduce
nums = [1, 2, 3, 4, 5]
print(list(map(lambda x: x*2, nums)))      # [2, 4, 6, 8, 10]
print(list(filter(lambda x: x%2==0, nums)))  # [2, 4]

from functools import reduce
print(reduce(lambda a, b: a+b, nums))  # 15

8. OOP Basics

class Animal:
    def __init__(self, name, sound):
        self.name = name
        self.sound = sound

    def speak(self):
        return f"{self.name} says {self.sound}"

class Dog(Animal):          # Inheritance
    def __init__(self, name):
        super().__init__(name, "Woof")

    def fetch(self):
        return f"{self.name} fetches the ball"

class Cat(Animal):
    def __init__(self, name):
        super().__init__(name, "Meow")

    def speak(self):        # Polymorphism (method overriding)
        return f"{self.name} purrs"

# Usage
d = Dog("Rex")
c = Cat("Whiskers")
print(d.speak())    # Rex says Woof
print(c.speak())    # Whiskers purrs
print(d.fetch())    # Rex fetches the ball

# isinstance check
print(isinstance(d, Animal))  # True
print(isinstance(d, Dog))     # True
print(isinstance(d, Cat))     # False

9. Common GATE Python Output Questions

Q1: What is the output?

x = [1, 2, 3]
y = x
y.append(4)
print(x)
Answer: [1, 2, 3, 4] — Lists are mutable. y = x creates a reference, not a copy. Both x and y point to the same list object.

Q2: What is the output?

def f(a, lst=[]):
    lst.append(a)
    return lst

print(f(1))
print(f(2))
print(f(3))
Answer: [1], [1, 2], [1, 2, 3] — Default mutable arguments are shared across calls! The list is created once when the function is defined.

Q3: What is the output?

print(type(3/2))
print(type(3//2))
print(3**2**2)
Answer: <class 'float'>, <class 'int'>, 81. Division always returns float. Floor division returns int. Exponentiation is right-associative: 3**(2**2) = 3**4 = 81.

Q4: What is the output?

a = [1, 2, [3, 4]]
b = a[:]            # shallow copy
b[2][0] = 99
print(a)
print(b)
Answer: [1, 2, [99, 4]] and [1, 2, [99, 4]]. Shallow copy copies outer list but inner list is still shared. Use import copy; b = copy.deepcopy(a) for full independence.

Q5: What is the output?

for i in range(3):
    pass
print(i)
Answer: 2. The loop variable persists after the loop ends in Python. The last value assigned was 2.

Q6: What is the output?

print([i*j for i in range(1,4) for j in range(1,4) if i != j])
Answer: [2, 3, 2, 6, 3, 6]. Pairs: (1,2)=2, (1,3)=3, (2,1)=2, (2,3)=6, (3,1)=3, (3,2)=6.
